         ;; Call the Python script and capture the index
         (col-index (string-to-number

    (if (< col-index 0)
        (error "Could not find 'lastRead' column in CSV header!")
      
      (save-excursion
        (beginning-of-line)
        ;; Move forward by the number of commas Python found
